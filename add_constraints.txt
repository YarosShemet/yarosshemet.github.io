CREATE TABLE address (
id number CONSTRAINT ad_id_un UNIQUE,
country VARCHAR(25),
city VARCHAR(15)
);

CREATE TABLE friends (
id number,
name VARCHAR(15),
email VARCHAR(25),
address_id number constraint ad_id_fk references address(id) ON DELETE SET NULL,
birthday VARCHAR(25),
CHECK (LENGTH(name)>3)
);
